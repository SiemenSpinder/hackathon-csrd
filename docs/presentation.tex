\section{Overview}
% ================================================================

\begin{frame}{What does the system do?}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Input}\\[4pt]
      An annual report (CSRD / Universal Registration Document) in Markdown format.

      \vspace{10pt}
      \textbf{Output (JSON)}
      \begin{itemize}
        \item 22 structured ESRS E2--E5 disclosure extractions with evidence \& metrics
        \item 5-dimension Nature Transition Plan (NTP) score modelled on the \textbf{WWF NAT40 framework}
        \item Weighted total score \textbf{out of 100}
      \end{itemize}

    \column{0.4\textwidth}
      \begin{block}{Why it matters}
        CSRD mandates disclosure of nature-related impacts (ESRS E2--E5). The WWF NAT40 study
        shows that only 1 in 40 French large-caps currently meets the average on Nature
        Transition Plans. This tool makes scoring fast, repeatable, and auditable.
      \end{block}
  \end{columns}
\end{frame}

% ================================================================

\begin{frame}{Pipeline Architecture}
  \centering
  \resizebox{\textwidth}{!}{%
  \begin{tikzpicture}[

      box/.style={rectangle, rounded corners=4pt, draw=wwfgreen, fill=wwfgreen!10,
                  text width=2.0cm, align=center, minimum height=0.7cm, font=\footnotesize},
      arrow/.style={-{Stealth[length=5pt]}, thick, color=wwfgreen},
    ]

    \node[box] (input) {Annual Report\\(Markdown)};
    \node[box, right=of input] (clean) {Clean \&\\Chunk Text};
    \node[box, right=of clean] (llm) {Azure OpenAI\\(Structured Output)};
    \node[box, right=of llm] (merge) {Merge\\Chunks};
    \node[box, right=of merge] (enrich) {Enrich \&\\Score};
    \node[box, right=of enrich] (output) {output.json};

    \draw[arrow] (input) -- (clean);
    \draw[arrow] (clean) -- (llm);
    \draw[arrow] (llm) -- (merge);
    \draw[arrow] (merge) -- (enrich);
    \draw[arrow] (enrich) -- (output);

  \end{tikzpicture}
  }%

  \vspace{8pt}
  \begin{columns}
    \column{0.33\textwidth}
      \textbf{Chunking strategy}\\
      Documents $>$100k tokens are split into overlapping chunks of 100k tokens
      with a 200-token overlap to avoid cutting mid-sentence.
    \column{0.33\textwidth}
      \textbf{Single LLM call per chunk}\\
      One \texttt{parse()} call returns \emph{both} ESRS disclosures and NTP scores,
      keeping latency and cost minimal.
    \column{0.33\textwidth}
      \textbf{Merge strategy}\\
      Disclosures: OR on \texttt{is\_present}, MAX on confidence, union of evidence/metrics.
      NTP scores: MAX per dimension across chunks.
  \end{columns}
\end{frame}

\section{NTP Scoring --- WWF NAT40 Framework}
% ================================================================

\begin{frame}{Nature Transition Plan (NTP) Scoring}
  \scriptsize
  Based on the \textbf{WWF NAT40 methodology} (Feb 2026) applied to CAC 40 companies.
  Five components scored on a \textbf{4-level maturity scale}:

  \vspace{6pt}
  % Levels table scaled to width using tabularx
  \begin{tabularx}{\textwidth}{c l X}
    \toprule
    \textbf{Level} & \textbf{Label} & \textbf{Meaning} \\
    \midrule
    0 & Non-aligned  & Absent or problematic disclosure \\
    1 & Compliant    & Meets minimum CSRD/ESRS transparency requirements \\
    2 & Coherent     & Exceeds minimum; integrates WWF/EFRAG expectations \\
    3 & Credible     & Advanced maturity; science-based frameworks (SBTN, TNFD) \\
    \bottomrule
  \end{tabularx}

  \vspace{8pt}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Weighted total score /100:}\\[4pt]
      \begin{tabularx}{\columnwidth}{l c c}
        \toprule
        Dimension & Weight & Max pts \\
        \midrule
        Foundations             & 25\% & 25 \\
        Metrics \& Targets      & 20\% & 20 \\
        Implementation Strategy & 30\% & 30 \\
        Engagement Strategy     & 15\% & 15 \\
        Governance              & 10\% & 10 \\
        \midrule
        \textbf{Total}          & 100\% & 100 \\
        \bottomrule
      \end{tabularx}
    \column{0.42\textwidth}
      \begin{block}{Formula}
        \[
          \text{Score} = \sum_{d} \frac{s_d}{3} \times w_d
        \]
        where $s_d \in \{0,1,2,3\}$ and $w_d$ is the dimension weight (in points).
      \end{block}
      \vspace{4pt}
      \footnotesize Implementation Strategy carries the \textbf{highest weight} to
      prioritise operational practices over disclosure alone (per WWF NAT40 design).
  \end{columns}
\end{frame}

\begin{frame}[shrink=5]{NTP Dimensions --- Scoring Criteria}
  \scriptsize
  \begin{tabularx}{\textwidth}{p{2.9cm} X X}
    \toprule
    \textbf{Dimension} & \textbf{Score 1--2 (Compliant/Coherent)} & \textbf{Score 3 (Credible)} \\
    \midrule
    \textbf{Foundations}
      & Double materiality covers direct ops; TNFD or SRTN methodology referenced
      & Multi-scale analysis (local/regional/global); IPBES pressure categories; KBAs mapped; aligned with Kunming-Montreal \\
    \addlinespace
    \textbf{Metrics \& Targets}
      & ESRS E2--E5 indicators for direct ops; targets with time horizons
      & Science-based targets (SBTN-aligned) or contextual/territorial targets linked to state-of-nature indicators \\
    \addlinespace
    \textbf{Implementation Strategy}
      & Action plans exist; linked to material impacts; some financial commitments
      & Site/supply-chain-specific plans; mitigation hierarchy (AR3T); formal financing plan (CAPEX/OPEX) \\
    \addlinespace
    \textbf{Engagement Strategy}
      & Named stakeholder groups; described mechanisms
      & Indigenous \& local community guarantees; lobbying positions disclosed \& aligned with nature commitments \\
    \addlinespace
    \textbf{Governance}
      & Board oversight documented; senior owner identified; nature in key functions
      & Nature KPIs in executive variable remuneration; formal training programmes for board \& employees \\
    \bottomrule
  \end{tabularx}
\end{frame}

\section{Data Models}
% ================================================================

\begin{frame}[fragile,shrink=10]{Pydantic Data Models}
  \begin{columns}
    \column{0.48\textwidth}
      \begin{lstlisting}[language=Python, title=Disclosure models]
class Evidence(BaseModel):
    quote: str

class MetricValue(BaseModel):
    name: str
    value: float
    unit: str
    year: Optional[int]

class DisclosureExtraction(BaseModel):
    code: ESRSCode        # E2-1 .. E5-6
    is_present: bool
    confidence: float     # [0,1]
    evidence: List[Evidence]
    metrics: List[MetricValue]
      \end{lstlisting}
    \column{0.48\textwidth}
      \begin{lstlisting}[language=Python, title=NTP scoring models]
class NTPDimensionScore(BaseModel):
    score: int   # 0-3
    rationale: str
    evidence: List[Evidence]

class NTPScoringResult(BaseModel):
    foundations: NTPDimensionScore
    metrics_and_targets: NTPDimensionScore
    implementation_strategy: NTPDimensionScore
    engagement_strategy: NTPDimensionScore
    governance: NTPDimensionScore

class CombinedExtractionResult(BaseModel):
    disclosures: List[DisclosureExtraction]
    ntp_scoring: NTPScoringResult
      \end{lstlisting}
  \end{columns}
  \vspace{4pt}
  \centering
  \footnotesize
  \texttt{CombinedExtractionResult} is passed directly as \texttt{response\_format}
  to \texttt{client.beta.chat.completions.parse()}, enforcing strict JSON output.
\end{frame}
